$c | replaceRE `\{\{<-?\s*img\b[^>]*\bsrc="([^"]+)"[^>]*\bcaption="([^"]+)"[^>]*>\}\}` "![$2]($1)" -}} {{- $c = $c | replaceRE `\{\{<-?\s*img\b[^>]*\bcaption="([^"]+)"[^>]*\bsrc="([^"]+)"[^>]*>\}\}` "![$1]($2)" -}} {{- $c = $c | replaceRE `\{\{<-?\s*img\b[^>]*\bsrc="([^"]+)"[^>]*>\}\}` "![image]($1)" -}} {{- $c = $c | replaceRE `\{\{<-?\s*link\s+"([^"]+)"\s+"([^"]+)"[^>]*>\}\}` "[$2]($1)" -}} {{- $c = $c | replaceRE `\{\{<-?\s*link\s+"([^"]+)"[^>]*>\}\}` "$1" -}} {{- $c = $c | replaceRE `\{\{<-?\s*download\s+"([^"]+)"\s+"([^"]+)"[^>]*>\}\}` "[Download: $2]($1)" -}} {{- $c = $c | replaceRE `\{\{<-?\s*download\s+"([^"]+)"[^>]*>\}\}` "[Download]($1)" -}} {{- $c = $c | replaceRE `\{\{<-?\s*(?:ref|relref)\s+"([^"]+)"[^>]*>\}\}` "$1" -}} {{- $c = $c | replaceRE `\{\{<-?\s*ghcode\s+"([^"]+)"[^>]*>\}\}` "\n*[Code: $1]*\n" -}} {{- $c = $c | replaceRE `\{\{<-?\s*ghcode\b[^>]*\burl="([^"]+)"[^>]*>\}\}` "\n*[Code: $1]*\n" -}} {{- $c = $c | replaceRE `\{\{<-?\s*call-out\s+"([^"]+)"[^>]*>\}\}` "**$1:** " -}} {{- $c = $c | replaceRE `\{\{<-?\s*call-out[^>]*>\}\}` "**Note:** " -}} {{- $c = $c | replaceRE `\{\{<-?\s*/call-out[^>]*>\}\}` "" -}} {{- $c = $c | replaceRE `\{\{<-?\s*important[^>]*>\}\}` "**Important:** " -}} {{- $c = $c | replaceRE `\{\{<-?\s*/important[^>]*>\}\}` "" -}} {{- $c = $c | replaceRE `\{\{<-?\s*warning[^>]*>\}\}` "**Warning:** " -}} {{- $c = $c | replaceRE `\{\{<-?\s*/warning[^>]*>\}\}` "" -}} {{- $c = $c | replaceRE `\{\{<-?\s*caution[^>]*>\}\}` "**Caution:** " -}} {{- $c = $c | replaceRE `\{\{<-?\s*/caution[^>]*>\}\}` "" -}} {{- $c = $c | replaceRE `\{\{<-?\s*deprecated[^>]*>\}\}` "**Deprecated:** " -}} {{- $c = $c | replaceRE `\{\{<-?\s*/deprecated[^>]*>\}\}` "" -}} {{- $c = $c | replaceRE `\{\{<-?\s*banner\s+"([^"]+)"[^>]*>\}\}` "**$1:** " -}} {{- $c = $c | replaceRE `\{\{<-?\s*banner[^>]*>\}\}` "**Note:** " -}} {{- $c = $c | replaceRE `\{\{<-?\s*/banner[^>]*>\}\}` "" -}} {{- $c = $c | replaceRE `\{\{<-?\s*tab\b[^>]*\bname="([^"]+)"[^>]*>\}\}` "\n#### $1\n" -}} {{- $c = $c | replaceRE `\{\{<-?\s*/tab[^>]*>\}\}` "\n" -}} {{- $c = $c | replaceRE
