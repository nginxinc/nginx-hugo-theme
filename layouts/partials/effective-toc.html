{{- /* effective-toc.html Returns the table-of-contents HTML for a page, accounting for contentSource. Hugo builds .TableOfContents only from a page's own headings, so a page that delegates its body via contentSource would otherwise show no (or only its own) headings. This merges the sourced page's headings first, followed by the delegating page's own headings, mirroring how single.html/single.md render the combined body. Param: a page object. Returns a string of TOC HTML. */ -}} {{- $page := . -}} {{- $ownToc := string $page.TableOfContents -}} {{- $sourceToc := "" -}} {{- with $page.Params.contentSource -}} {{- with site.GetPage . -}} {{- $sourceToc = string .TableOfContents -}} {{- end -}} {{- end -}} {{- $result := $ownToc -}} {{- if $sourceToc -}} {{- $extract := `(?s)^.*?
\s*
    (.*)
\s*
.*$` -}} {{- $sourceItems := "" -}} {{- if in $sourceToc "
  • " -}}{{- $sourceItems = replaceRE $extract "$1" $sourceToc -}}{{- end -}} {{- $ownItems := "" -}} {{- if in $ownToc "
  • " -}}{{- $ownItems = replaceRE $extract "$1" $ownToc -}}{{- end -}} {{- if or $sourceItems $ownItems -}} {{- $result = printf `
      %s%s
    ` $sourceItems $ownItems -}} {{- else -}} {{- $result = "" -}} {{- end -}} {{- end -}} {{- return $result -}}
